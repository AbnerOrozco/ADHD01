para el siguiente ejemplo lo hicimos de la sigueinte manera


utilizando @for, @each y @while

$colors: (
red,
blue,
green,
yellow,
orange,
purple,
pink,
brown,
grey,
black
);

$iterations: 10;

%box-base {
padding: 20px;
border-radius: 5px;
text-align: center;
margin: 10px auto;
width: 80%;
font-size: 20px;
color: white;
transition: background-color 0.3s, color 0.3s;
}

@for $i from 1 through $iterations {
.box-#{$i} {
@extend %box-base;
font-size: 16px + $i * 2;
}
}

@each $color in $colors {
.box-#{$color} {
background-color: $color;
}
}

$i: 1;
@while $i <= length($colors) {
.box-#{$i} {
background-color: nth($colors, $i);
}
$i: $i + 1;
}
